# 3. App.jsx

```jsx
import { useEffect, useState } from 'react'
import Header from './components/Header'
import Navbar from './components/Navbar'
import Sidebar from './components/Sidebar'
import Programmes from './components/Programmes'
import Research from './components/Research'
import Footer from './components/Footer'

function App() {

  const [departmentData, setDepartmentData] = useState(null)

  useEffect(() => {

    fetch('/data/department.xml')
      .then((response) => response.text())
      .then((xmlText) => {

        const parser = new DOMParser()
        const xml = parser.parseFromString(xmlText, 'text/xml')

        const title = xml.querySelector('title').textContent

        const descriptions = [...xml.querySelectorAll('info description')]
          .map((item) => item.textContent)

        const programmes = [...xml.querySelectorAll('programme')].map((item) => ({
          name: item.querySelector('name').textContent,
          icon: item.querySelector('icon').textContent,
          details: item.querySelector('details').textContent,
        }))

        const research = [...xml.querySelectorAll('field')].map((item) => ({
          title: item.querySelector('title').textContent,
          description: item.querySelector('description').textContent,
        }))

        setDepartmentData({
          title,
          descriptions,
          programmes,
          research,
        })
      })

  }, [])

  if (!departmentData) {
    return <h2 style={{ padding: '40px' }}>Loading...</h2>
  }

  return (
    <>

      <Header />

      <Navbar />

      <div className="main-wrap">

        <Sidebar />

        <div className="content">

          <h1>{departmentData.title}</h1>

          <img
            src="/images/cse_dept_pic.jpg"
            alt="Department"
            className="department-image"
          />

          {
            departmentData.descriptions.map((text, index) => (
              <p key={index}>{text}</p>
            ))
          }

          <Programmes programmes={departmentData.programmes} />

          <Research research={departmentData.research} />

        </div>

      </div>

      <Footer />

    </>
  )
}

export default App
```

# 11. Install & Run

## Create React App using Vite

```bash
npm create vite@latest cse-portal -- --template react
```

## Go into folder

```bash
cd cse-portal
```

## Install dependencies

```bash
npm install
```

## Start server

```bash
npm run dev
```

---

# Why this version is Dynamic

This version is dynamic because:

* XML stores all department data.
* React fetches and parses XML automatically.
* Components are reusable.
* Programmes and research sections render dynamically using `.map()`.
* Changing XML updates the website automatically.
* Layout is component-based and scalable.
